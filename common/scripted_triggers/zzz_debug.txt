		AND = {
			tag = ENG

					type = submarine
					size > 30
				}
			}
		}
		AND = {
			tag = USA
			not = { is_in_faction_with = GER }
			GER = {
				has_navy_size = {
					type = submarine
					size > 30
				}
			}
		}
		AND = {
			tag = JAP

		}
		AND = {
			has_war = yes

				has_navy_size = {
					type = submarine

				}
			}
		}

	}
